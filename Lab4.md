#!/bin/bash

# add, subtract, multiply, divide 2 numeric value
((result1 = 1+100))
((result2 = 100-1))
((result3 = 9*9))
((result4 = 9/9))
# Print the results
echo $result1
echo $result2
echo $result3
echo $result4
